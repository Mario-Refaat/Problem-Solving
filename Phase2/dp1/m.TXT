#include <bits/stdc++.h>
using namespace std;
#define ll long long

int main() {

	ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);

	int t = 1;
	//cin >> t;

	while (t--) {

		ll n, x;
		cin >> n;
		vector<ll>v;
		while (n--) {
			cin >> x;
			if (v.empty() || v.back() < x) v.push_back(x);
			else v[lower_bound(v.begin(), v.end(), x) - v.begin()] = x;
		}
		cout << v.size();

	}

	return 0;
}
