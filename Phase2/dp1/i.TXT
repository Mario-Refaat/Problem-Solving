#include <bits/stdc++.h>
using namespace std;
#define ll long long

ll n, m, dp[5005][5005];
vector<ll>v;

ll f(ll i = 0, ll lst = 0) {

	if (i == n) return 0;

	ll& ret = dp[i][lst];
	if (~ret) return ret;

	ret = 1e18;
	if (v[i] >= lst) ret = min(ret, f(i + 1, v[i]));
	ret = min(ret, f(i + 1, lst) + 1);

	return ret;
}

int main() {

	ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);

	int t = 1;
	//cin >> t;

	while (t--) {

		memset(dp, -1, sizeof dp);

		cin >> n >> m;
		double x;
		v.resize(n);
		for (int i = 0; i < n; i++) cin >> v[i] >> x;

		cout << f();
	}

	return 0;
}
