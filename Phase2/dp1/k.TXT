#include <bits/stdc++.h>
using namespace std;
#define ll long long

ll rec(vector<ll>& v, ll l, ll r, ll k , ll res = 0) {

	if (res > k || l > r) return res;
	if (v[l] == v[r]) return rec(v, l + 1, r - 1, k, res);
	
	return min(rec(v, l + 1, r, k, res + 1), rec(v, l, r - 1, k, res + 1));
}

int main() {

	ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);

	int t = 1;
	cin >> t;

	for(int tt = 1 ; tt<=t;tt++) {

		ll n, k;
		cin >> n >> k;

		vector<ll>v(n);
		for (int i = 0; i < n; i++) cin >> v[i];

		ll res = rec(v, 0, n - 1, k);

		cout << "Case " << tt << ": ";
		if (res == 0) cout << "Too easy" << "\n";
		else if (res > k) cout << "Too difficult" << "\n";
		else cout << res << "\n";

	}

	return 0;
}
