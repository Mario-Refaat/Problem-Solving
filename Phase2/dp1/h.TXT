#include <bits/stdc++.h>
using namespace std;
#define ll long long

int main() {

	ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);

	int t = 1;
	//cin >> t;

	while (t--) {

		ll n, mx = -1e18;
		cin >> n;

		vector<vector<ll>>v(n + 1, vector<ll>(n + 1));

		for (int i = 1; i <= n; i++) {
			for (int j = 1; j <= n; j++) {
				cin >> v[i][j]; v[i][j] += v[i - 1][j] + v[i][j - 1] - v[i - 1][j - 1];
			}
		}

		for (int i = 1; i <= n; i++) {
			for (int j = 1; j <= n; j++) {
				for (int x = i; x <= n; x++) {
					for (int y = j; y <= n; y++) {
						mx = max(mx, v[x][y] - v[i - 1][y] - v[x][j - 1] + v[i - 1][j - 1]);
					}
				}
			}
		}

		cout << mx <<"\n";
	}

	return 0;
}
