#include <bits/stdc++.h>
using namespace std;
#define ll long long

ll n, n5, n10, dp[151][500][55], total;
ll f(ll i, ll n5, ll n10) {

	if (i == n) return 0;

	ll& ret = dp[i][n5][n10];
	if (~ret) return ret;

	ret = 1e18;
		
	ll n1 = total - (i * 8 + n5 * 5 + n10 * 10);
	if (n1 >= 8) ret = min(ret, f(i + 1, n5, n10) + 8);
	if (n5 >= 2) ret = min(ret, f(i + 1, n5 - 2, n10) + 2);
	if (n5 >= 1 && n1 >= 3) ret = min(ret, f(i + 1, n5 - 1, n10) + 4);
	if (n10 >= 1) ret = min(ret, f(i + 1, n5, n10 - 1) + 1);
	if (n10 >= 1 && n1 >= 3) ret = min(ret, f(i + 1, n5 + 1, n10 - 1) + 4);

	return ret;
}

int main() {

	ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);

	int t = 1;
	cin >> t;

	while (t--) {

		memset(dp, -1, sizeof dp);

		cin >> n >> n1 >> n5 >> n10;

		total = n1 + n5 * 5 + n10 * 10;
		cout << f(0, n5, n10) << "\n";

	}

	return 0;
}
