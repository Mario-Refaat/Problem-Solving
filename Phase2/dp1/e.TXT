#include <bits/stdc++.h>
using namespace std;
#define ll long long

ll n, m, k, dp[105][105][105];
vector<ll>v;
vector<vector<ll>>cost;

ll f(ll i = 0, ll lst = 0, ll rem = k) {

	if (i == n) return (rem == 0 ? 0 : 1e18);
	if (rem < 0) return 1e18;

	ll& ret = dp[i][lst][rem];
	if (~ret) return ret;

	ret = 1e18;
	if (v[i] == 0) {
		for (int j = 1; j <= m; j++)
			if (j == lst) ret = min(ret, f(i + 1, lst, rem) + cost[i][j]);
			else ret = min(ret, f(i + 1, j, rem - 1) + cost[i][j]);
	}
	else if (v[i] == lst) ret = f(i + 1, lst, rem);
	else ret = f(i + 1, v[i], rem - 1);

	return ret;
}

int main() {

	ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);

	int t = 1;
	//cin >> t;

	while (t--) {

		memset(dp, -1, sizeof dp);

		cin >> n >> m >> k;
		v.resize(n);
		cost.resize(n, vector<ll>(m + 1));

		for (int i = 0; i < n; i++)cin >> v[i];
		for (int i = 0; i < n; i++)
			for (int j = 1; j <= m; j++)
				cin >> cost[i][j];

		ll ans = f();
		cout << (ans == 1e18 ? -1 : ans);
	}

	return 0;
}
