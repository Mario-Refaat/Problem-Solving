#include <bits/stdc++.h>
using namespace std;
#define ll long long

ll n, m, dp[2005][2005] , md = 1e9+7;
vector<string>v;

ll f(ll i = 0, ll j = 0) {

	if (i == n - 1 && j == m - 1) return 1;
	if (i >= n || j >= m || v[i][j] == '#') return 0;

	ll& ret = dp[i][j];
	if (~ret) return ret;

	return ret = (f(i + 1, j)%md + f(i, j + 1)%md)%md;
}

int main() {

	ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);

	int t = 1;
	//cin >> t;

	while (t--) {

		memset(dp, -1, sizeof dp);
		cin >> n >> m;

		v.resize(n);
		for (int i = 0; i < n; i++) cin >> v[i];

		cout << f();
	}

	return 0;
}
