#include <bits/stdc++.h>
using namespace std;
#define ll long long

ll n, m , dp[200005];
vector<vector<ll>>v;

ll f(ll node) {

	if (v[node].empty()) return 0;
	
	ll& ret = dp[node];
	if (~ret) return ret;

	ret = 0;
	for (auto e : v[node]) ret = max(ret,f(e)+1);

	return ret;
}

int main() {

	ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);

	int t = 1;
	//cin >> t;

	while (t--) {

		memset(dp, -1, sizeof dp);
		cin >> n >> m;

		v.resize(n+1);

		ll x, y;
		while (m--) {
			cin >> x >> y;
			v[x].push_back(y);
		}

		ll ans = 0;
		for (int i = 1; i <= n; i++) ans = max(ans, f(i));

		cout << ans;
	}

	return 0;
}
