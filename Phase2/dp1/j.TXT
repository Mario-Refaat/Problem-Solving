#include <bits/stdc++.h>
using namespace std;
#define ll long long

ll n, m, k, dp[51][5005];
vector<ll>v;
string s;

ll f(ll i , ll sum) {

	if (sum <= 0) return 0;

	ll& ret = dp[i][sum];
	if (~ret) return ret;

	ret = 1e18;
	for (int j = 0; j < n; j++) 
		if (v[j] > v[i] && s[i] != s[j]) ret = min(ret, f(j, sum - v[j]) + abs(i - j));
	
	return ret;
}

int main() {

	ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);

	int t = 1;
	//cin >> t;

	while (t--) {

		memset(dp, -1, sizeof dp);

		cin >> n >> m >> k;
		v.resize(n);
		for (int i = 0; i < n; i++) cin >> v[i];
		cin >> s;

		ll ans = 1e18;
		for (int i = 0; i < n; i++) {
			ans = min(ans, f(i, k - v[i]) + abs(i - (m - 1)));
		}

		cout << (ans == 1e18 ? -1 : ans);
	}

	return 0;
}
